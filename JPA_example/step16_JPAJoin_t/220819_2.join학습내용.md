create table Team (
       team_id bigint not null auto_increment,
        team_name varchar(255),
        primary key (team_id)
    ) engine=InnoDB
    
create table Member (
       member_id bigint not null auto_increment,
        name varchar(255),
        team_id bigint,
        primary key (member_id)
    ) engine=InnoDB
